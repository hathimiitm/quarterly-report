Sample: Calculating NPV (Python)


def npv(cashflows, discount_rate):
    pv = 0.0
    for t, cf in enumerate(cashflows, start=1):
        pv += cf / (1 + discount_rate) ** t
    return pv

cashflows = [100, 100, 100, 100, 100]
print(npv(cashflows, 0.08))
